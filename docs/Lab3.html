death_data <- colorado %>%
  mutate(year = year(date)) %>%  
  filter(year == 2021) %>%
  group_by(fips) %>%
  summarize(deaths = sum(new_deaths, na.rm = TRUE)) %>%
  left_join(pop, by = "fips") %>%  
  mutate(death_ratio = 100 * (deaths / DEATHS2021)) %>%
  select(CTYNAME, deaths, DEATHS2021, death_ratio) %>%
  filter(death_ratio > 20)
